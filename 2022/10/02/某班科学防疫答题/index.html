一、 登录取身份cookie

# name: 用户名
# pwd: 密码
# return: cookie字典
def login(name,pwd):
    usr_name = name
    usr_psw = pwd
    session = requests.session()
    login_parameter = {
        'a': 'l',
        '_u': usr_name,
        '_s': base64.b64encode(usr_psw.encode()).decode(),
        '_c': '0',
        '_p': 'true'
    }
    response = session.get('http://zz.yiban.cn/zxkc/user?a=l',
                           params=login_parameter,
                           headers=headers).json()

    if not response["status"] is True:
        raise Exception(response["data"])
    return session.cookies.get_dict()

def sg(cookies):
