from unicodedata import combining
import requests
import time
import hashlib

headers = {
    'User-Agent':
    'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.0.0 Safari/537.36'
}

url = "http://zz.yiban.cn/topic/2022/selfbelay"

def sg(cookies):
    para = {
    'a':'sg',
    '_time': int(round(time.time() * 1000))
    }
    # 加密串
    response = requests.get(url,params=para,cookies=cookies,headers=headers).json()
    if not response["status"] is True:
        raise Exception(response["data"])
    print(response['data'])
    return response["data"]


def post(dat,cookies):
    data = str(dat) + "_6"
    ck = hashlib.md5(data.encode("utf-8")).hexdigest()
    para = {
        'a':'dg',
        'uc':'6',
        'ck': ck,
        '_time': int(round(time.time() * 1000))
    }
    # 告知成功
    response = requests.post(url,cookies=cookies,headers=headers,params=para).json()
    if not response["status"] is True:
        raise Exception(response["data"])
    print(response["data"])

def raffle(cookies):
    para = {
        'a':'dl',

    }

    try:
