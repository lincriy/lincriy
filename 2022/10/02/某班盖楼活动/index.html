某班盖楼活动

发表于2022-10-02|更新于2023-04-03
|字数总计:270|阅读时长:1分钟

cookie获取调用了《某班科学防疫答题》中的登录代码

完整代码示例

# filename: building
import requests
import time
import hashlib

headers = {
    'User-Agent':
    'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.0.0 Safari/537.36'
}

def sg(cookies):
    time_res = int(round(time.time() * 1000))
    url = f"http://zz.yiban.cn/topic/2022/semester?a=sg&_time={time_res}"

    response = requests.request("GET", url, headers=headers,
                                cookies=cookies).json()
    if not response["status"] is True:
        raise Exception(response["data"])
    data = str(response['data'])
    print(data)
    return data


def dg(res,sc,cookies):
    data = str(res) + str(sc)
    ck = hashlib.md5(data.encode("utf-8")).hexdigest()
    time_res = int(round(time.time() * 1000))
    url = f"http://zz.yiban.cn/topic/2022/semester?a=dg&uc={sc}&ck={ck}&_time={time_res}"

    response = requests.request("POST", url, headers=headers,
                                cookies=cookies).json()
    if not response["status"] is True:
        raise Exception(response["data"])
    print(response["data"])

def dl(cookies):
    time_res = int(round(time.time() * 1000))
    url = f"http://zz.yiban.cn/topic/2022/semester?a=dl&_time={time_res}"

    response = requests.request("GET", url, headers=headers,
                                cookies=cookies).json()
    if not response["status"] is True:
        raise Exception(response["data"])
    print(response['data'])

def combine_work(cookie,floor):
    try:
        se_data = sg(cookie)
        input("等待")
        dg(se_data,floor,cookie)
        dl(cookie)
    except Exception as e:
        print(e)

# 此处填入自己的cookie
combine_work(cookie,25)
