读EXCEL

  • 提取EXCEL中某列数据
public static ArrayList<String> excelReader() {
    // 结果集
    ArrayList<String> names = new ArrayList<>();
    // try-resource，读项目路径文件
    try(InputStream inputStream = new  FileInputStream("src/main/resources/UserNames.xlsx");
        // 根据文件新建文档对象
        XSSFWorkbook workbook = new XSSFWorkbook(inputStream)){
        // 取文档中的第一个工作表
        XSSFSheet sheet0 = workbook.getSheetAt(0);
        // 遍历行，把每行的第5个的内容放到结果集中
        sheet0.rowIterator().forEachRemaining(row -> {
            Cell cell = row.getCell(4);
            names.add(cell.getStringCellValue());
        });
    }catch(IOException e){
        e.printStackTrace();
    }
    // 返回结果集
    return names;
}

读PDF

  • 提取PDF中所有字符内容
public static String pdfReader(){
    PdfReader pdfReader = null;
    // 结果集
    StringBuilder builder = new StringBuilder();
    try {
        // 根据输入流，构造PdfReader实例
        pdfReader = new PdfReader("src/main/resources/1.pdf");
        // getNumberOfPages方法获取总页数
        for (int i = 1; i <= pdfReader.getNumberOfPages(); i++) {
            // 获取每一页的内容
            String text = PdfTextExtractor.getTextFromPage(pdfReader, i);
            // 添加到结果集中
            builder.append(text).append("\n--------");
        }
    } catch (IOException e) {
        throw new RuntimeException(e);
    } finally {
        if (pdfReader != null) {
            pdfReader.close();
        }
    }
    return builder.toString();
}

水平切割图片

/**
 * 横向图片切割：把一个纵向的长图水平切割成多个小图，每张小图保存在src/main/resources/目录下
 * @param heightForEachImage 小图片的高度
*/
public static void HorizontalImageCut(String sourceFile,int heightForEachImage){
    File inputFile = new File(sourceFile);
    String sourceFileFolder = inputFile.getParent();
    String fileSuffix = inputFile.toString().substring(inputFile.toString().lastIndexOf('.'));
    System.out.println( sourceFileFolder);
    System.out.println("hello");
    System.out.println("hello");
    try(FileInputStream inputStream = new FileInputStream(sourceFile)){
        // 读原图
        BufferedImage image = ImageIO.read(inputStream);
        int width = image.getWidth();
        int height = heightForEachImage;
        // 计算实际需要切割的小图数量
        int imageCount = (int) Math.ceil(((double) image.getHeight()) /height);
        // 结果图片集合
        BufferedImage[] bufferedImages = new BufferedImage[imageCount];
        // 绘制每一张结果图
        for (int i = 0; i < imageCount; i++) {
            // 把每一张结果图的宽度设定为原图宽度，高度设定为要求的小图高度（若是最后一张图则取剩余高度）
            bufferedImages[i] = new BufferedImage(width,i == imageCount - 1 ? image.getHeight()-i*height: height,image.getType());
            // 新建2D画布
            Graphics2D graphics = bufferedImages[i].createGraphics();
            // 从image中截出小图，放到bufferedImages[i]中
            graphics.drawImage(image,0,0,width,height,0,i*height,width,(i+1)*height,null);
        }
        // 写出图片到指定文件夹
        for (int i = 0; i < bufferedImages.length; i++) {
            ImageIO.write(bufferedImages[i],"jpg",new File(sourceFileFolder + "/" + i + fileSuffix));
        }
    } catch (IOException e) {
        throw new RuntimeException(e);
    }
}

百度OCR识别文字

/***
 * 调用百度OCR接口识别图片文字内容(每个月免费1000次)：结果打印在控制台并写入到源目录下的result.json文件中方便后续使用
 * @param sourceFileFolder 存放源图片的目录
 * @param MaxImageIndex 总图片个数，若{@code suffix}指定的值是.jpg，函数会遍历0.jpg、1.jpg、...、{@code MaxImageIndex}.jpg
 * @param suffix 图片的后缀，读取的图片会以该格式为后缀
*/
public static void OCRImageReader(String sourceFileFolder,Integer MaxImageIndex,String suffix) {
    // 初始化一个AipOcr，填入自己账号中的三个参数
    String APP_ID = "APP_ID";
    String API_KEY = "API_KEY";
    String SECRET_KEY = "SECRET_KEY";
    AipOcr client = new AipOcr(APP_ID, API_KEY, SECRET_KEY);

    // 可选：设置网络连接参数
    client.setConnectionTimeoutInMillis(2000);
    client.setSocketTimeoutInMillis(60000);
    // 创建结果集
    JSONArray jsonArray = new JSONArray();
    // 对每一张图片调用一次接口
    for (int i = 0; i < MaxImageIndex+1; i++) {
        // 拼接出图片路径
        String path = sourceFileFolder + "/" + + i + suffix;
        // 断言是一个文件
        assert new File(path).isFile();
        // 调用接口识别文字，返回JSON对象
        JSONObject res = client.basicGeneral(path, new HashMap<>());
        // 把识别结果添加到结果集中
        jsonArray.put(res);
    }
    try (FileWriter writer = new FileWriter(sourceFileFolder+"/"+"result.json")) {
        // 该处的toString的参数indentFactor是缩进因子，这样打印出来会更美观
        writer.append(jsonArray.toString(2));
    } catch (IOException e) {
        throw new RuntimeException(e);
    }
    // 再往控制台输出一份
    System.out.println(jsonArray.toString(2));

}
